<script>
  // Dados do quiz
  const questions = [
    { text: "Quando nos conhecemos?", answer: "31/10", hint: "Dia e mês (ex: 31/10)", type: "input" },
    { text: "Qual o momento mais marcante nosso?", answer: "Nossa primeira viagem juntos sozinhos.", options: ["O nosso primeiro beijo.", "Nossa primeira viagem juntos sozinhos.", "A primeira vez que disse \"eu te amo\".", "O dia em que nos conhecemos."], type: "choice" },
    { text: "Qual a nossa maior qualidade?", answer: "O amor um pelo outro", options: ["A paciência infinita.", "O amor um pelo outro", "O senso de humor.", "A cumplicidade nas aventuras."], type: "choice" },
    { text: "Qual a nossa maior loucura?", answer: "Viver a nossa vida, sem ligar para ninguém.", options: ["Viajar sem planejar nada.", "Viver a nossa vida, sem ligar para ninguém.", "Ficar acordados a noite inteira conversando.", "Fazer promessas malucas de amor."], type: "choice" },
    { text: "Aceita dividir essa vida maluca comigo novamente?", answer: "Com certeza", options: ["Com certeza", "Preciso pensar...", "Talvez...", "Só se tiver muito amor!"], type: "choice" }
  ];

  let currentQuestion = 0;
  let canAnswer = true;
  let timer;

  // Elementos DOM
  const welcomeScreen = document.getElementById('welcome-screen');
  const quizScreen = document.getElementById('quiz-screen');
  const celebrationScreen = document.getElementById('celebration-screen');
  const progressDots = document.getElementById('progress-dots');
  const qNumber = document.getElementById('q-number');
  const qText = document.getElementById('q-text');
  const answerArea = document.getElementById('answer-area');
  const feedbackDiv = document.getElementById('feedback');

  // Função para criar pétalas
  function createPetals() {
    const container = document.getElementById('petal-bg');
    const petalSymbols = ['🌸', '🌹', '🩷', '🌺', '✿'];
    for (let i = 0; i < 35; i++) {
      const petal = document.createElement('div');
      petal.className = 'petal';
      petal.textContent = petalSymbols[Math.floor(Math.random() * petalSymbols.length)];
      petal.style.left = Math.random() * 100 + '%';
      petal.style.fontSize = (Math.random() * 20 + 12) + 'px';
      petal.style.animationDuration = (Math.random() * 12 + 8) + 's';
      petal.style.animationDelay = (Math.random() * 15) + 's';
      petal.style.opacity = '0.6';
      container.appendChild(petal);
    }
  }
  createPetals();

  // Função para atualizar os dots
  function updateDots() {
    progressDots.innerHTML = '';
    for (let i = 0; i < questions.length; i++) {
      const dot = document.createElement('div');
      dot.className = 'dot';
      if (i === currentQuestion) dot.classList.add('active');
      if (i < currentQuestion) dot.classList.add('completed');
      progressDots.appendChild(dot);
    }
  }

  // Normalizar texto
  function normalize(str) {
    return str.trim().toLowerCase().replace(/\s+/g, ' ').replace(/[.,!?;]/g, '');
  }

  // Exibir feedback correto
  function showCorrect() {
    feedbackDiv.innerHTML = '<div class="text-[#d4a54a] text-lg font-semibold">✨ Perfeita, meu amor! ✨</div>';
  }

  function showWrong() {
    const msgs = ["Não foi dessa vez, tente de novo coração 🌹", "Quase lá, amor! Pensa com carinho ❤️", "Essa não é a resposta certa, mas confio em você 💕", "Tenta outra vez, meu bem! Estamos quase lá 💖"];
    feedbackDiv.innerHTML = `<div class="text-[#ff9f9f] text-base">${msgs[Math.floor(Math.random() * msgs.length)]}</div>`;
  }

  // Renderizar pergunta atual
  function renderQuestion() {
    const q = questions[currentQuestion];
    qNumber.textContent = `Pergunta ${currentQuestion + 1} de ${questions.length}`;
    qText.textContent = q.text;
    answerArea.innerHTML = '';
    feedbackDiv.innerHTML = '';
    canAnswer = true;

    if (q.type === 'input') {
      const wrapper = document.createElement('div');
      wrapper.className = 'flex flex-col items-center gap-4';
      const input = document.createElement('input');
      input.type = 'text';
      input.placeholder = q.hint || 'Digite sua resposta...';
      input.className = 'quiz-input';
      const btn = document.createElement('button');
      btn.textContent = 'Responder 💝';
      btn.className = 'btn-primary mt-2';
      btn.onclick = () => handleInput(input.value);
      input.addEventListener('keypress', (e) => { if (e.key === 'Enter') handleInput(input.value); });
      wrapper.appendChild(input);
      wrapper.appendChild(btn);
      answerArea.appendChild(wrapper);
      setTimeout(() => input.focus(), 100);
    } else {
      const grid = document.createElement('div');
      grid.className = 'flex flex-col gap-3';
      q.options.forEach(opt => {
        const btn = document.createElement('button');
        btn.textContent = opt;
        btn.className = 'quiz-option';
        btn.onclick = () => handleChoice(opt, btn, grid);
        grid.appendChild(btn);
      });
      answerArea.appendChild(grid);
    }
    updateDots();
  }

  function handleInput(value) {
    if (!canAnswer) return;
    if (!value.trim()) return;
    const current = questions[currentQuestion];
    if (normalize(value) === normalize(current.answer)) {
      canAnswer = false;
      showCorrect();
      disableInputs(true);
      setTimeout(() => nextQuestion(), 1400);
    } else {
      showWrong();
      const inputField = document.querySelector('#answer-area input');
      if (inputField) { inputField.value = ''; inputField.focus(); }
    }
  }

  function handleChoice(selected, btn, grid) {
    if (!canAnswer) return;
    const current = questions[currentQuestion];
    const allBtns = grid.querySelectorAll('button');
    allBtns.forEach(b => b.style.pointerEvents = 'none');
    if (normalize(selected) === normalize(current.answer)) {
      canAnswer = false;
      btn.style.background = 'rgba(212,165,74,0.3)';
      btn.style.borderColor = '#d4a54a';
      showCorrect();
      setTimeout(() => nextQuestion(), 1400);
    } else {
      btn.style.background = 'rgba(255,80,80,0.2)';
      btn.style.borderColor = '#ff7777';
      showWrong();
      setTimeout(() => {
        allBtns.forEach(b => b.style.pointerEvents = 'auto');
        btn.style.background = '';
        btn.style.borderColor = 'rgba(196,35,72,0.4)';
      }, 1300);
    }
  }

  function disableInputs(disable) {
    const btns = document.querySelectorAll('#answer-area button, #answer-area input');
    btns.forEach(el => {
      if (disable) el.style.pointerEvents = 'none';
      else el.style.pointerEvents = 'auto';
    });
  }

  function nextQuestion() {
    if (currentQuestion + 1 < questions.length) {
      currentQuestion++;
      renderQuestion();
    } else {
      showCelebration();
    }
  }

  // Celebração final com confetes e corações
  function showCelebration() {
    // Mostrar tela de celebração
    welcomeScreen.classList.remove('active');
    quizScreen.classList.remove('active');
    celebrationScreen.classList.add('active');

    // Efeitos
    for (let i = 0; i < 120; i++) {
      setTimeout(() => {
        const conf = document.createElement('div');
        conf.className = 'confetti';
        const colors = ['#c42348', '#d4a54a', '#ff6f91', '#ffb347', '#f08080'];
        conf.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
        conf.style.width = (Math.random() * 8 + 4) + 'px';
        conf.style.height = (Math.random() * 8 + 4) + 'px';
        conf.style.left = Math.random() * 100 + '%';
        conf.style.top = '-10px';
        conf.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';
        conf.style.animation = `confettiFall ${Math.random() * 2 + 2}s linear forwards`;
        document.body.appendChild(conf);
        setTimeout(() => conf.remove(), 3000);
      }, i * 35);
    }

    // Corações flutuantes
    for (let i = 0; i < 40; i++) {
      setTimeout(() => {
        const heart = document.createElement('div');
        heart.className = 'floating-heart';
        heart.textContent = ['❤️', '💖', '💗', '🌹', '💕'][Math.floor(Math.random() * 5)];
        heart.style.left = Math.random() * 80 + 10 + '%';
        heart.style.fontSize = (Math.random() * 20 + 20) + 'px';
        document.body.appendChild(heart);
        setTimeout(() => heart.remove(), 3500);
      }, i * 120);
    }
  }

  // Adicionar estilo para confete
  const style = document.createElement('style');
  style.textContent = `
    @keyframes confettiFall {
      0% { transform: translateY(0) rotate(0deg); opacity: 1; }
      100% { transform: translateY(100vh) rotate(720deg); opacity: 0; }
    }
  `;
  document.head.appendChild(style);

  // Controle de telas
  function showScreen(screenId) {
    welcomeScreen.classList.remove('active');
    quizScreen.classList.remove('active');
    celebrationScreen.classList.remove('active');
    document.getElementById(screenId).classList.add('active');
  }

  // Eventos
  document.getElementById('start-btn').addEventListener('click', () => {
    currentQuestion = 0;
    showScreen('quiz-screen');
    renderQuestion();
  });

  document.getElementById('restart-btn').addEventListener('click', () => {
    currentQuestion = 0;
    showScreen('welcome-screen');
    // Limpar qualquer estado residual
    if (timer) clearTimeout(timer);
    canAnswer = true;
  });
</script>
